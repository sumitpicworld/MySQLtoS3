-- Create Database
CREATE DATABASE IF NOT EXISTS your_project_database;
USE your_project_database;

-- Create Users Table
CREATE TABLE IF NOT EXISTS users (
    user_id INT PRIMARY KEY,
    username VARCHAR(50) NOT NULL,
    email VARCHAR(100) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Create Orders Table
CREATE TABLE IF NOT EXISTS orders (
    order_id INT PRIMARY KEY,
    user_id INT,
    product_name VARCHAR(255) NOT NULL,
    quantity INT,
    order_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(user_id)
);

-- Query to Retrieve User and Order Information
SELECT users.user_id, users.username, users.email, orders.order_id, orders.product_name, orders.quantity
FROM users
JOIN orders ON users.user_id = orders.user_id;
